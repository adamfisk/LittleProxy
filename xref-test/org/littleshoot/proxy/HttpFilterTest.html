34      @Test
35      public void testFiltering() throws Exception {
36  
37          final AtomicInteger shouldFilterCalls = new AtomicInteger(0);
38          final AtomicInteger filterCalls = new AtomicInteger(0);
39          final Queue<HttpRequest> associatedRequests =
40                  new LinkedList<HttpRequest>();
41  
42          final String url1 = "http://localhost:8924/";
43          final String url2 = "http://localhost:8924/testing";
44          final HttpFilter filter = new HttpFilter() {
45  
46              public boolean filterResponses(final HttpRequest httpRequest) {
47                  shouldFilterCalls.incrementAndGet();
48                  return true;
49              }
50  
51              public int getMaxResponseSize() {
52                  return 1024 * 1024;
53              }
54  
55              public void filterResponse(final HttpRequest httpRequest,
56                      final HttpResponse response) {
57                  filterCalls.incrementAndGet();
58                  if (httpRequest != null) {
59                      associatedRequests.add(httpRequest);
60                  } else {
61                      log.error("REQUEST IS NULL!!");
62                  }
63              }
64          };
65          final HttpResponseFilters responseFilters =
66                  new HttpResponseFilters() {
67                      public HttpFilter getFilter(final String hostAndPort) {
68                          if (hostAndPort.equals("localhost:8924")) {
69                              return filter;
70                          }
71                          return null;
72                      }
73                  };
74          final HttpProxyServer server = DefaultHttpProxyServer.bootstrap()
75                  .withResponseFilters(responseFilters)
76                  .create().listen(PROXY_PORT);
77          boolean connected = false;
78          final InetSocketAddress isa = new InetSocketAddress("127.0.0.1",
79                  PROXY_PORT);
80          while (!connected) {
81              final Socket sock = new Socket();
82              try {
83                  sock.connect(isa);
84                  break;
85              } catch (final IOException e) {
86                  // Keep trying.
87              } finally {
88                  IOUtils.closeQuietly(sock);
89              }
90              Thread.sleep(50);
91          }
92  
93          final Server webServer = new Server(WEB_SERVER_PORT);
94          webServer.start();
95  
96          getResponse(url1);
97  
98          assertEquals(1, associatedRequests.size());
99          assertEquals(1, shouldFilterCalls.get());
100         assertEquals(1, filterCalls.get());
101 
102         // We just open a second connection here since reusing the original
103         // connection is inconsistent.
104         getResponse(url2);
105 
106         assertEquals(2, shouldFilterCalls.get());
107         assertEquals(2, filterCalls.get());
108         assertEquals(2, associatedRequests.size());
109 
110         final HttpRequest first = associatedRequests.remove();
111         final HttpRequest second = associatedRequests.remove();
112 
113         // Make sure the requests in the filter calls were the requests they
114         // actually should have been.
115         assertEquals(url1, first.getUri());
116         assertEquals(url2, second.getUri());

118         webServer.stop();

120     }

122     private HttpEntity getResponse(final String url) throws Exception {
123         final DefaultHttpClient http = new DefaultHttpClient();
124         final HttpHost proxy = new HttpHost("127.0.0.1", PROXY_PORT, "http");
125         http.getParams().setParameter(ConnRoutePNames.DEFAULT_PROXY, proxy);
126         final HttpGet get = new HttpGet(url);
127         final org.apache.http.HttpResponse hr = http.execute(get);
128         final HttpEntity responseEntity = hr.getEntity();
129         EntityUtils.consume(responseEntity);
130         http.getConnectionManager().shutdown();
131         return responseEntity;
