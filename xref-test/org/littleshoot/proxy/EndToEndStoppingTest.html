47      /**
48       * This is a quick test from nasis that exhibits different behavior from
49       * unit tests because unit tests call System.exit(). The stop method should
50       * stop all non-daemon threads and should cause the JVM to exit without
51       * explicitly calling System.exit(), which running as an application
52       * properly tests.
53       */
54      public static void main(final String[] args) throws Exception {
55          int port = 9090;
56          HttpProxyServer proxyServer = DefaultHttpProxyServer.bootstrap()
57                  .withPort(port)
58                  .start();
59  
60          Proxy proxy = new Proxy();
61          proxy.setProxyType(Proxy.ProxyType.MANUAL);
62          String proxyStr = String.format("localhost:%d", port);
63          proxy.setHttpProxy(proxyStr);
64          proxy.setSslProxy(proxyStr);
65  
66          DesiredCapabilities capability = DesiredCapabilities.firefox();
67          capability.setCapability(CapabilityType.PROXY, proxy);
68  
69          String urlString = "http://www.yahoo.com/";
70          WebDriver driver = new FirefoxDriver(capability);
71          driver.manage().timeouts().pageLoadTimeout(30, TimeUnit.SECONDS);
72  
73          driver.get(urlString);
74  
75          driver.close();
76          System.out.println("Driver closed");
77  
78          proxyServer.stop();
79          System.out.println("Proxy stopped");
80      }
