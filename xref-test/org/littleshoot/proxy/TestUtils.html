57      /**
58       * Creates and starts embedded web server that is running on given port,
59       * including an SSL connector on the other given port. Each response has a
60       * body that indicates how many bytes were received with a message like
61       * "Received x bytes\n".
62       * 
63       * @param port
64       *            The port
65       * @param sslPort
66       *            (optional) The ssl port
67       * @return Instance of Server
68       * @throws Exception
69       *             if failed to start
70       */
71      public static Server startWebServer(final int port, final Integer sslPort)
72              throws Exception {
73          final Server httpServer = new Server(port);
74          httpServer.setHandler(new AbstractHandler() {
75              public void handle(String target, Request baseRequest,
76                      HttpServletRequest request, HttpServletResponse response)
77                      throws IOException, ServletException {
78                  if (request.getRequestURI().contains("hang")) {
79                      System.out.println("Hanging as requested");
80                      try {
81                          Thread.sleep(90000);
82                      } catch (InterruptedException ie) {
83                          System.out.println("Stopped hanging due to interruption");
84                      }
85                  }
86                  
87                  long numberOfBytesRead = 0;
88                  InputStream in = new BufferedInputStream(request
89                          .getInputStream());
90                  while (in.read() != -1) {
91                      numberOfBytesRead += 1;
92                  }
93                  System.out.println("Done reading # of bytes: "
94                          + numberOfBytesRead);
95                  response.setStatus(HttpServletResponse.SC_OK);
96                  baseRequest.setHandled(true);
97                  byte[] content = ("Received " + numberOfBytesRead + " bytes\n").getBytes();
98                  response.addHeader("Content-Length", Integer.toString(content.length));
99                  response.getOutputStream().write(content);
100             }
101         });
102         if (sslPort != null) {
103             // Add SSL connector
104             org.eclipse.jetty.util.ssl.SslContextFactory sslContextFactory = new org.eclipse.jetty.util.ssl.SslContextFactory();
105 
106             SelfSignedSslEngineSource contextSource = new SelfSignedSslEngineSource();
107             SSLContext sslContext = contextSource.getSslContext();
108 
109             sslContextFactory.setSslContext(sslContext);
110             SslSocketConnector connector = new SslSocketConnector(
111                     sslContextFactory);
112             connector.setPort(sslPort);
113             /*
114              * <p>Ox: For some reason, on OS X, a non-zero timeout can causes
115              * sporadic issues. <a href="http://stackoverflow.com/questions
116              * /16191236/tomcat-startup-fails
117              * -due-to-java-net-socketexception-invalid-argument-on-mac-o">This
118              * StackOverflow thread</a> has some insights into it, but I don't
119              * quite get it.</p>
120              * 
121              * <p>This can cause problems with Jetty's SSL handshaking, so I
122              * have to set the handshake timeout and the maxIdleTime to 0 so
123              * that the SSLSocket has an infinite timeout.</p>
124              */
125             connector.setHandshakeTimeout(0);
126             connector.setMaxIdleTime(0);
127             httpServer.addConnector(connector);
128         }
129         httpServer.start();
130         return httpServer;
131     }

146     /**
147      * Creates instance HttpClient that is configured to use proxy server. The
148      * proxy server should run on 127.0.0.1 and given port
149      * 
150      * @param port
151      *            the proxy port
152      * @param supportSSL
153      *            if true, client will support SSL connections to servers using
154      *            self-signed certificates
155      * @return instance of HttpClient
156      */
157     public static HttpClient createProxiedHttpClient(final int port,
158             final boolean supportSSL) throws Exception {
159         final HttpClient httpclient = new DefaultHttpClient();
160         // Note: we use 127.0.0.1 here because on OS X, using straight up
161         // localhost yields a connect exception.
162         final HttpHost proxy = new HttpHost("127.0.0.1", port, "http");
163         httpclient.getParams().setParameter(ConnRoutePNames.DEFAULT_PROXY,
164                 proxy);
165         if (supportSSL) {
166             SSLSocketFactory sf = new SSLSocketFactory(
167                     new TrustSelfSignedStrategy());
168             sf.setHostnameVerifier(new X509HostnameVerifier() {
169                 public boolean verify(String arg0, SSLSession arg1) {
170                     return true;
171                 }
172 
173                 public void verify(String host, String[] cns,
174                         String[] subjectAlts) throws SSLException {
175                 }
176 
177                 public void verify(String host, X509Certificate cert)
178                         throws SSLException {
179                 }
180 
181                 public void verify(String host, SSLSocket ssl)
182                         throws IOException {
183                 }
184             });
185             Scheme scheme = new Scheme("https", 443, sf);
186             httpclient.getConnectionManager().getSchemeRegistry()
187                     .register(scheme);
188         }
189         return httpclient;
190     }

192     public static int randomPort() {
193         final SecureRandom secureRandom = new SecureRandom();
194         for (int i = 0; i < 20; i++) {
195             // The +1 on the random int is because
196             // Math.abs(Integer.MIN_VALUE) == Integer.MIN_VALUE -- caught
197             // by FindBugs.
198             final int randomPort = 1024 + (Math.abs(secureRandom.nextInt() + 1) % 60000);
199             ServerSocket sock = null;
200             try {
201                 sock = new ServerSocket();
202                 sock.bind(new InetSocketAddress("127.0.0.1", randomPort));
203                 final int port = sock.getLocalPort();
204                 return port;
205             } catch (final IOException e) {
206             } finally {
207                 if (sock != null) {
208                     try {
209                         sock.close();
210                     } catch (IOException e) {
211                     }
212                 }
213             }
214         }
215 
216         // If we can't grab one of our securely chosen random ports, use
217         // whatever port the OS assigns.
218         ServerSocket sock = null;
219         try {
220             sock = new ServerSocket();
221             sock.bind(null);
222             final int port = sock.getLocalPort();
223             return port;
224         } catch (final IOException e) {
225             return 1024 + (Math.abs(secureRandom.nextInt() + 1) % 60000);
226         } finally {
227             if (sock != null) {
228                 try {
229                     sock.close();
230                 } catch (IOException e) {
231                 }
232             }
233         }
234     }

246                 + "\nMax NH :" + muNH.getMax() + "\nUsed NH:" + muNH.getUsed()
247                 + "\nCommitted NH:" + muNH.getCommitted());

249         // Below courtesy of:
250         // http://neopatel.blogspot.com/2011/05/java-count-open-file-handles.html
251         OperatingSystemMXBean osStats = ManagementFactory
252                 .getOperatingSystemMXBean();
253         long numberOfOpenFileDescriptors = 0;
254         if (osStats.getClass().getName()
255                 .equals("com.sun.management.UnixOperatingSystem")) {
256             Method method = osStats.getClass().getDeclaredMethod(
257                     "getOpenFileDescriptorCount");
258             method.setAccessible(true);
259             numberOfOpenFileDescriptors = (Long) method.invoke(osStats);
260             System.out.println("Open File Descriptors: "
261                     + numberOfOpenFileDescriptors);
262             method.setAccessible(false);
263         }
264         return numberOfOpenFileDescriptors;
