52      /**
53       * Creates and starts embedded web server that is running on given port,
54       * including an SSL connector on the other given port. Each response has a
55       * body that indicates how many bytes were received with a message like
56       * "Received x bytes\n".
57       * 
58       * @param port
59       *            The port
60       * @param sslPort
61       *            (optional) The ssl port
62       * @return Instance of Server
63       * @throws Exception
64       *             if failed to start
65       */
66      public static Server startWebServer(final int port, final Integer sslPort)
67              throws Exception {
68          final Server httpServer = new Server(port);
69          httpServer.setHandler(new AbstractHandler() {
70              public void handle(String target, Request baseRequest,
71                      HttpServletRequest request, HttpServletResponse response)
72                      throws IOException, ServletException {
73                  long numberOfBytesRead = 0;
74                  InputStream in = new BufferedInputStream(request
75                          .getInputStream());
76                  while (in.read() != -1) {
77                      numberOfBytesRead += 1;
78                  }
79                  System.out.println("Done reading # of bytes: "
80                          + numberOfBytesRead);
81                  response.setStatus(HttpServletResponse.SC_OK);
82                  baseRequest.setHandled(true);
83                  byte[] content = ("Received " + numberOfBytesRead + " bytes\n").getBytes();
84                  response.addHeader("Content-Length", Integer.toString(content.length));
85                  response.getOutputStream().write(content);
86              }
87          });
88          if (sslPort != null) {
89              // Add SSL connector
90              org.eclipse.jetty.util.ssl.SslContextFactory sslContextFactory = new org.eclipse.jetty.util.ssl.SslContextFactory();
91  
92              SelfSignedSSLEngineSource contextSource = new SelfSignedSSLEngineSource();
93              SSLContext sslContext = contextSource.getSslContext();
94  
95              sslContextFactory.setSslContext(sslContext);
96              SslSocketConnector connector = new SslSocketConnector(
97                      sslContextFactory);
98              connector.setPort(sslPort);
99              /*
100              * <p>Ox: For some reason, on OS X, a non-zero timeout can causes
101              * sporadic issues. <a href="http://stackoverflow.com/questions
102              * /16191236/tomcat-startup-fails
103              * -due-to-java-net-socketexception-invalid-argument-on-mac-o">This
104              * StackOverflow thread</a> has some insights into it, but I don't
105              * quite get it.</p>
106              * 
107              * <p>This can cause problems with Jetty's SSL handshaking, so I
108              * have to set the handshake timeout and the maxIdleTime to 0 so
109              * that the SSLSocket has an infinite timeout.</p>
110              */
111             connector.setHandshakeTimeout(0);
112             connector.setMaxIdleTime(0);
113             httpServer.addConnector(connector);
114         }
115         httpServer.start();
116         return httpServer;
117     }

132     /**
133      * Creates instance HttpClient that is configured to use proxy server. The
134      * proxy server should run on 127.0.0.1 and given port
135      * 
136      * @param port
137      *            the proxy port
138      * @param supportSSL
139      *            if true, client will support SSL connections to servers using
140      *            self-signed certificates
141      * @return instance of HttpClient
142      */
143     public static HttpClient createProxiedHttpClient(final int port,
144             final boolean supportSSL) throws Exception {
145         final HttpClient httpclient = new DefaultHttpClient();
146         // Note: we use 127.0.0.1 here because on OS X, using straight up
147         // localhost yields a connect exception.
148         final HttpHost proxy = new HttpHost("127.0.0.1", port, "http");
149         httpclient.getParams().setParameter(ConnRoutePNames.DEFAULT_PROXY,
150                 proxy);
151         if (supportSSL) {
152             SSLSocketFactory sf = new SSLSocketFactory(
153                     new TrustSelfSignedStrategy());
154             sf.setHostnameVerifier(new X509HostnameVerifier() {
155                 public boolean verify(String arg0, SSLSession arg1) {
156                     return true;
157                 }
158 
159                 public void verify(String host, String[] cns,
160                         String[] subjectAlts) throws SSLException {
161                 }
162 
163                 public void verify(String host, X509Certificate cert)
164                         throws SSLException {
165                 }
166 
167                 public void verify(String host, SSLSocket ssl)
168                         throws IOException {
169                 }
170             });
171             Scheme scheme = new Scheme("https", 443, sf);
172             httpclient.getConnectionManager().getSchemeRegistry()
173                     .register(scheme);
174         }
175         return httpclient;
176     }

178     public static int randomPort() {
179         final SecureRandom secureRandom = new SecureRandom();
180         for (int i = 0; i < 20; i++) {
181             // The +1 on the random int is because
182             // Math.abs(Integer.MIN_VALUE) == Integer.MIN_VALUE -- caught
183             // by FindBugs.
184             final int randomPort = 1024 + (Math.abs(secureRandom.nextInt() + 1) % 60000);
185             ServerSocket sock = null;
186             try {
187                 sock = new ServerSocket();
188                 sock.bind(new InetSocketAddress("127.0.0.1", randomPort));
189                 final int port = sock.getLocalPort();
190                 return port;
191             } catch (final IOException e) {
192             } finally {
193                 if (sock != null) {
194                     try {
195                         sock.close();
196                     } catch (IOException e) {
197                     }
198                 }
199             }
200         }
201 
202         // If we can't grab one of our securely chosen random ports, use
203         // whatever port the OS assigns.
204         ServerSocket sock = null;
205         try {
206             sock = new ServerSocket();
207             sock.bind(null);
208             final int port = sock.getLocalPort();
209             return port;
210         } catch (final IOException e) {
211             return 1024 + (Math.abs(secureRandom.nextInt() + 1) % 60000);
212         } finally {
213             if (sock != null) {
214                 try {
215                     sock.close();
216                 } catch (IOException e) {
217                 }
218             }
219         }
220     }
