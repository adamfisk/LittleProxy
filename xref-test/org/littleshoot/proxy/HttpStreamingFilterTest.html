188     private org.apache.http.HttpResponse getResponse(final String url)
189             throws Exception {
190         final DefaultHttpClient http = new DefaultHttpClient();
191         final HttpHost proxy = new HttpHost("127.0.0.1", PROXY_PORT, "http");
192         http.getParams().setParameter(ConnRoutePNames.DEFAULT_PROXY, proxy);
193         final HttpGet get = new HttpGet(url);
194         final org.apache.http.HttpResponse hr = http.execute(get);
195         final HttpEntity responseEntity = hr.getEntity();
196         EntityUtils.consume(responseEntity);
197         http.getConnectionManager().shutdown();
198         return hr;
