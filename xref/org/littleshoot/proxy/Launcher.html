21   * Launches a new HTTP proxy.

23  public class Launcher {

25      private static final Logger LOG = LoggerFactory.getLogger(Launcher.class);

27      private static final String OPTION_DNSSEC = "dnssec";
28  
29      private static final String OPTION_PORT = "port";
30  
31      private static final String OPTION_HELP = "help";
32  
33      /**
34       * Starts the proxy from the command line.
35       * 
36       * @param args
37       *            Any command line arguments.
38       */
39      public static void main(final String... args) {
40          pollLog4JConfigurationFileIfAvailable();
41          LOG.info("Running LittleProxy with args: {}", Arrays.asList(args));
42          final Options options = new Options();
43          options.addOption(null, OPTION_DNSSEC, true,
44                  "Request and verify DNSSEC signatures.");
45          options.addOption(null, OPTION_PORT, true, "Run on the specified port.");
46          options.addOption(null, OPTION_HELP, false,
47                  "Display command line help.");
48          final CommandLineParser parser = new PosixParser();
49          final CommandLine cmd;
50          try {
51              cmd = parser.parse(options, args);
52              if (cmd.getArgs().length > 0) {
53                  throw new UnrecognizedOptionException(
54                          "Extra arguments were provided in "
55                                  + Arrays.asList(args));
56              }
57          } catch (final ParseException e) {
58              printHelp(options,
59                      "Could not parse command line: " + Arrays.asList(args));
60              return;
61          }
62          if (cmd.hasOption(OPTION_HELP)) {
63              printHelp(options, null);
64              return;
65          }
66          final int defaultPort = 8080;
67          int port;
68          if (cmd.hasOption(OPTION_PORT)) {
69              final String val = cmd.getOptionValue(OPTION_PORT);
70              try {
71                  port = Integer.parseInt(val);
72              } catch (final NumberFormatException e) {
73                  printHelp(options, "Unexpected port " + val);
74                  return;
75              }
76          } else {
77              port = defaultPort;
78          }
79  
80          System.out.println("About to start server on port: " + port);
81          HttpProxyServerBootstrap bootstrap = DefaultHttpProxyServer
82                  .bootstrapFromFile("./littleproxy.properties")
83                  .withPort(port)
84                  .withAllowLocalOnly(false);
85  
86          if (cmd.hasOption(OPTION_DNSSEC)) {
87              final String val = cmd.getOptionValue(OPTION_DNSSEC);
88              if (ProxyUtils.isTrue(val)) {
89                  LOG.info("Using DNSSEC");
90                  bootstrap.withUseDnsSec(true);
91              } else if (ProxyUtils.isFalse(val)) {
92                  LOG.info("Not using DNSSEC");
93                  bootstrap.withUseDnsSec(false);
94              } else {
95                  printHelp(options, "Unexpected value for " + OPTION_DNSSEC
96                          + "=:" + val);
97                  return;
98              }
99          }
100 
101         System.out.println("About to start...");
102         bootstrap.start();
103     }

105     private static void printHelp(final Options options,
106             final String errorMessage) {
107         if (!StringUtils.isBlank(errorMessage)) {
108             LOG.error(errorMessage);
109             System.err.println(errorMessage);
110         }
111 
112         final HelpFormatter formatter = new HelpFormatter();
113         formatter.printHelp("littleproxy", options);
114     }

119         if (log4jConfigurationFile.exists()) {
