7   /**
8    * <p>
9    * A helper class that logs messages for ProxyConnections. All it does is make
10   * sure that the Channel and current state are always included in the log
11   * messages (if available).
12   * </p>
13   * 
14   * <p>
15   * Note that this depends on us using a LocationAwareLogger so that we can
16   * report the line numbers of the caller rather than this helper class.
17   * </p>
18   */
19  class ProxyConnectionLogger {
20      private final ProxyConnection connection;
21      private final LocationAwareLogger logger;
22      private final String fqcn = this.getClass().getCanonicalName();
23  
24      public ProxyConnectionLogger(ProxyConnection connection) {
25          this.connection = connection;
26          this.logger = (LocationAwareLogger) LoggerFactory.getLogger(connection
27                  .getClass());
28      }
29  
30      protected void error(String message, Object... params) {
31          if (logger.isErrorEnabled()) {
32              doLog(LocationAwareLogger.ERROR_INT, message, params, null);
33          }
34      }
35  
36      protected void error(String message, Throwable t) {
37          if (logger.isErrorEnabled()) {
38              doLog(LocationAwareLogger.ERROR_INT, message, null, t);
39          }
40      }
41  
42      protected void warn(String message, Object... params) {
43          if (logger.isWarnEnabled()) {
44              doLog(LocationAwareLogger.WARN_INT, message, params, null);
45          }
46      }
47  
48      protected void warn(String message, Throwable t) {
49          if (logger.isWarnEnabled()) {
50              doLog(LocationAwareLogger.WARN_INT, message, null, t);
51          }
52      }
53  
54      protected void info(String message, Object... params) {
55          if (logger.isInfoEnabled()) {
56              doLog(LocationAwareLogger.INFO_INT, message, params, null);
57          }
58      }
59  
60      protected void info(String message, Throwable t) {
61          if (logger.isInfoEnabled()) {
62              doLog(LocationAwareLogger.INFO_INT, message, null, t);
63          }
64      }
65  
66      protected void debug(String message, Object... params) {
67          if (logger.isDebugEnabled()) {
68              doLog(LocationAwareLogger.DEBUG_INT, message, params, null);
69          }
70      }
71  
72      protected void debug(String message, Throwable t) {
73          if (logger.isDebugEnabled()) {
74              doLog(LocationAwareLogger.DEBUG_INT, message, null, t);
75          }
76      }
77  
78      private void doLog(int level, String message, Object[] params, Throwable t) {
79          String formattedMessage = fullMessage(message);
80          if (params != null && params.length > 0) {
81              formattedMessage = MessageFormatter.arrayFormat(formattedMessage,
82                      params).getMessage();
83          }
84          logger.log(null, fqcn, level, formattedMessage, null, t);
85      }
86  
87      private String fullMessage(String message) {
88          String stateMessage = connection.getCurrentState().toString();
89          if (connection.isTunneling()) {
90              stateMessage += " {tunneling}";
91          }
92          String messagePrefix = "(" + stateMessage + ")";
93          if (connection.channel != null) {
94              messagePrefix = messagePrefix + " " + connection.channel;
95          }
96          return messagePrefix + ": " + message;
97      }
